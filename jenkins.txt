pipeline {
    agent any

    environment {
        NODE_ENV = 'production'
        MONGO_URI = credentials('mongo-uri') // optional: used in your app, not Jenkins
    }

    stages {
        stage('Install') {
            steps {
                sh 'npm install'
            }
        }

        stage('Test') {
            steps {
                sh 'npm test'
            }
        }
    }

    post {
        success {
            slackSend (
                channel: '#YourFirstName_IP1',
                message: "✅ Build #${env.BUILD_ID} passed. Site is live at: https://your-render-url.onrender.com"
            )
        }

        failure {
            slackSend (
                channel: '#YourFirstName_IP1',
                message: "❌ Build #${env.BUILD_ID} failed."
            )
            mail to: 'your@email.com',
                 subject: "❌ Jenkins Build #${env.BUILD_ID} Failed",
                 body: "The Jenkins pipeline failed. Check Jenkins logs for more details."
        }
    }
}
